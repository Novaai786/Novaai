📘 Email Campaign Setup Guide (Google Apps Script)

This guide lets you send up to 500 free emails a day from your own Gmail account, directly from NovaAICRM — no paid tool needed. Setup takes about 5 minutes and you only do it once. You do not need to create a Google Sheet yourself — the script creates and manages its own sheet automatically.

Step 1: Open Apps Script

Go to script.google.com → New project. Delete any default code already in Code.gs.

Step 2: Paste This Code

// ===== NovaAICRM — Gmail Email Engine (Google Apps Script) =====
// No manual Google Sheet needed — this script creates and manages
// its own Google Sheet automatically the first time it runs.
// Handles: Send Now, Schedule for Later, Sequences, and Open Tracking.

const SHEET_NAME_QUEUE = 'EmailQueue';
const SHEET_NAME_LOG = 'EmailLog';

// ---------- Spreadsheet is created automatically (no setup needed) ----------
function getOrCreateSpreadsheet() {
  const props = PropertiesService.getScriptProperties();
  let sheetId = props.getProperty('SHEET_ID');
  if (sheetId) {
    try { return SpreadsheetApp.openById(sheetId); } catch (e) { /* recreate below */ }
  }
  const ss = SpreadsheetApp.create('NovaAICRM Email Data');
  props.setProperty('SHEET_ID', ss.getId());
  const queue = ss.getSheets()[0];
  queue.setName(SHEET_NAME_QUEUE);
  queue.appendRow(['queueId', 'to', 'subject', 'body', 'fromName', 'scheduledAt', 'status', 'sentAt']);
  const log = ss.insertSheet(SHEET_NAME_LOG);
  log.appendRow(['trackId', 'to', 'subject', 'fromName', 'status', 'sentAt', 'openedAt']);
  return ss;
}

function getSheet(name) {
  const ss = getOrCreateSpreadsheet();
  let sh = ss.getSheetByName(name);
  if (!sh) {
    sh = ss.insertSheet(name);
    if (name === SHEET_NAME_QUEUE) sh.appendRow(['queueId', 'to', 'subject', 'body', 'fromName', 'scheduledAt', 'status', 'sentAt']);
    if (name === SHEET_NAME_LOG) sh.appendRow(['trackId', 'to', 'subject', 'fromName', 'status', 'sentAt', 'openedAt']);
  }
  return sh;
}

// ---------- Background timer installs itself automatically ----------
function ensureTrigger() {
  const exists = ScriptApp.getProjectTriggers().some(t => t.getHandlerFunction() === 'processQueue');
  if (!exists) {
    ScriptApp.newTrigger('processQueue').timeBased().everyMinutes(10).create();
  }
}

// ---------- Helpers ----------
function genId() { return Utilities.getUuid(); }
function jsonOut(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); }
function trackingPixelTag(trackId) {
  const url = ScriptApp.getService().getUrl();
  return '<img src="' + url + '?track=' + trackId + '" width="1" height="1" style="display:none">';
}

// ---------- Main API used by the CRM ----------
function doPost(e) {
  ensureTrigger();
  let req;
  try { req = JSON.parse(e.postData.contents); }
  catch (err) { return jsonOut({ success: false, error: 'Invalid request' }); }

  try {
    switch (req.action) {
      case 'sendEmail': return handleSendEmail(req);
      case 'scheduleEmail': return handleScheduleEmail(req);
      case 'getQueue': return handleGetQueue();
      case 'cancelScheduled': return handleCancelScheduled(req);
      case 'deleteQueueItem': return handleDeleteQueueItem(req);
      case 'updateScheduled': return handleUpdateScheduled(req);
      case 'getStatus': return handleGetStatus();
      case 'deleteEmailLog': return handleDeleteEmailLog(req);
      default: return jsonOut({ success: false, error: 'Unknown action: ' + req.action });
    }
  } catch (err) {
    return jsonOut({ success: false, error: err.message });
  }
}

// ---------- Open-tracking pixel ----------
function doGet(e) {
  const trackId = e.parameter.track;
  if (trackId) {
    try {
      const sh = getSheet(SHEET_NAME_LOG);
      const data = sh.getDataRange().getValues();
      for (let i = 1; i < data.length; i++) {
        if (data[i][0] === trackId && !data[i][6]) {
          sh.getRange(i + 1, 5).setValue('Opened');
          sh.getRange(i + 1, 7).setValue(new Date());
          break;
        }
      }
    } catch (err) { /* silent */ }
  }
  const pixel = Utilities.base64Decode('R0lGODlhAQABAIAAAAAAAP///yH5BAEAAAAALAAAAAABAAEAAAIBTAA7');
  return ContentService.createBinaryOutput(pixel).setMimeType(ContentService.MimeType.GIF);
}

// ---------- Action handlers ----------
function handleSendEmail(req) {
  const trackId = genId();
  const fromName = req.fromName || 'NovaAICRM';
  try {
    const htmlBody = String(req.body).replace(/\n/g, '<br>') + trackingPixelTag(trackId);
    GmailApp.sendEmail(req.to, req.subject, req.body, { htmlBody: htmlBody, name: fromName });
    getSheet(SHEET_NAME_LOG).appendRow([trackId, req.to, req.subject, fromName, 'Sent', new Date(), '']);
    return jsonOut({ success: true, trackId: trackId });
  } catch (err) {
    return jsonOut({ success: false, error: err.message });
  }
}

function handleScheduleEmail(req) {
  const queueId = genId();
  getSheet(SHEET_NAME_QUEUE).appendRow([queueId, req.to, req.subject, req.body, req.fromName || 'NovaAICRM', req.scheduledAt, 'pending', '']);
  return jsonOut({ success: true, queueId: queueId });
}

function handleGetQueue() {
  const data = getSheet(SHEET_NAME_QUEUE).getDataRange().getValues();
  const rows = [];
  for (let i = 1; i < data.length; i++) {
    const r = data[i];
    rows.push({ queueId: r[0], to: r[1], subject: r[2], body: r[3], fromName: r[4], scheduledAt: r[5], status: r[6], sentAt: r[7] });
  }
  return jsonOut({ success: true, data: rows });
}

function findQueueRow(sh, queueId) {
  const data = sh.getDataRange().getValues();
  for (let i = 1; i < data.length; i++) { if (data[i][0] === queueId) return i + 1; }
  return -1;
}

function handleCancelScheduled(req) {
  const sh = getSheet(SHEET_NAME_QUEUE);
  const row = findQueueRow(sh, req.queueId);
  if (row === -1) return jsonOut({ success: false, error: 'Not found' });
  sh.getRange(row, 7).setValue('cancelled');
  return jsonOut({ success: true });
}

function handleDeleteQueueItem(req) {
  const sh = getSheet(SHEET_NAME_QUEUE);
  const row = findQueueRow(sh, req.queueId);
  if (row === -1) return jsonOut({ success: false, error: 'Not found' });
  sh.deleteRow(row);
  return jsonOut({ success: true });
}

function handleUpdateScheduled(req) {
  const sh = getSheet(SHEET_NAME_QUEUE);
  const row = findQueueRow(sh, req.queueId);
  if (row === -1) return jsonOut({ success: false, error: 'Not found' });
  sh.getRange(row, 3).setValue(req.subject);
  sh.getRange(row, 4).setValue(req.body);
  sh.getRange(row, 6).setValue(req.scheduledAt);
  return jsonOut({ success: true });
}

function handleGetStatus() {
  const data = getSheet(SHEET_NAME_LOG).getDataRange().getValues();
  const rows = [];
  for (let i = 1; i < data.length; i++) {
    const r = data[i];
    rows.push({ trackId: r[0], to: r[1], subject: r[2], fromName: r[3], status: r[4], sentAt: r[5], openedAt: r[6] });
  }
  return jsonOut({ success: true, data: rows });
}

function handleDeleteEmailLog(req) {
  const sh = getSheet(SHEET_NAME_LOG);
  const data = sh.getDataRange().getValues();
  for (let i = 1; i < data.length; i++) {
    if (data[i][0] === req.trackId) { sh.deleteRow(i + 1); break; }
  }
  return jsonOut({ success: true });
}

// ---------- Runs automatically every ~10 minutes: sends due scheduled/sequence emails ----------
function processQueue() {
  const sh = getSheet(SHEET_NAME_QUEUE);
  const data = sh.getDataRange().getValues();
  const now = new Date();
  for (let i = 1; i < data.length; i++) {
    const [queueId, to, subject, body, fromName, scheduledAt, status] = data[i];
    if (status === 'pending' && to && new Date(scheduledAt) <= now) {
      try {
        const trackId = genId();
        const htmlBody = String(body).replace(/\n/g, '<br>') + trackingPixelTag(trackId);
        GmailApp.sendEmail(to, subject, body, { htmlBody: htmlBody, name: fromName || 'NovaAICRM' });
        sh.getRange(i + 1, 7).setValue('sent');
        sh.getRange(i + 1, 8).setValue(new Date());
        getSheet(SHEET_NAME_LOG).appendRow([trackId, to, subject, fromName || 'NovaAICRM', 'Sent', new Date(), '']);
      } catch (err) {
        sh.getRange(i + 1, 7).setValue('failed: ' + err.message);
      }
      Utilities.sleep(500);
    }
  }
}

Nothing to edit in this code — no Spreadsheet ID, no manual sheet setup. It's ready to deploy as-is.

Step 3: Deploy as a Web App

Deploy → New deployment → click the gear icon (⚙️) → select Web app.
Execute as: Me  |  Who has access: Anyone
Click Deploy. If asked to authorize, click Authorize access → choose your Google account → Advanced → "Go to project (unsafe)" → Allow.
Copy the Web app URL you're given (it ends with /exec).

Step 4: Connect It to the CRM

Paste that Web app URL into "Sender Setup" above and click Save Settings. That's it — Send Campaign Now, Schedule for Later, and Sequences all work immediately. The background timer that sends scheduled and sequence emails installs itself automatically the first time you send or schedule anything — no separate step needed, and it checks every 10 minutes for anything due.

⚠️ Good to know:
  • Free Gmail limit: 500 emails/day (1,500/day on Google Workspace).
  • This script only ever uses your own Gmail account — no one's email or password is ever requested.
  • If you ever redeploy and get a new URL, remember to update it again in Sender Setup above.

📘 WhatsApp & SMS Setup Guide

This uses your own Twilio account. Twilio's free trial gives you a WhatsApp Sandbox and a trial SMS number to test with.

Step 1: Get Twilio credentials

Sign up at twilio.com → Console Dashboard → copy your Account SID and Auth Token. For WhatsApp, join your Sandbox (Messaging → Try it out → Send a WhatsApp message) to get a sandbox number like whatsapp:+14155238886. For SMS, use a Twilio phone number.

Step 2: Create the proxy script

Go to script.google.com → New project → delete the placeholder code → paste this:

function doPost(e) {
  const req = JSON.parse(e.postData.contents);
  try {
    if (req.action === 'sendWhatsApp' || req.action === 'sendSMS') {
      const url = 'https://api.twilio.com/2010-04-01/Accounts/' + req.accountSid + '/Messages.json';
      const payload = { To: req.to, From: req.from, Body: req.body };
      const options = {
        method: 'post',
        payload: payload,
        headers: { Authorization: 'Basic ' + Utilities.base64Encode(req.accountSid + ':' + req.authToken) },
        muteHttpExceptions: true
      };
      const response = UrlFetchApp.fetch(url, options);
      const result = JSON.parse(response.getContentText());
      if (result.sid) {
        return ContentService.createTextOutput(JSON.stringify({ success: true, sid: result.sid })).setMimeType(ContentService.MimeType.JSON);
      }
      return ContentService.createTextOutput(JSON.stringify({ success: false, error: result.message || 'Twilio error' })).setMimeType(ContentService.MimeType.JSON);
    }
    return ContentService.createTextOutput(JSON.stringify({ success: false, error: 'Unknown action' })).setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ success: false, error: err.message })).setMimeType(ContentService.MimeType.JSON);
  }
}

Step 3: Deploy as a Web App

Deploy → New deployment → gear icon ⚙️ → select Web app. Execute as: Me  |  Who has access: Anyone. Click Deploy, authorize it if asked, then copy the Web app URL (ends with /exec).

Step 4: Connect it here

Paste that URL into "Twilio Proxy Web App URL" above, along with your Account SID, Auth Token, and From numbers, then click Save Settings. You can now use "Send WhatsApp" / "Send SMS" actions in Workflow Automation, or the quick test box.

⚠️ Good to know:
  • Trial Twilio accounts can only message numbers you've verified, and WhatsApp Sandbox recipients must first text your sandbox a join code.
  • Your Account SID/Auth Token are stored in your own Supabase project and are only ever sent to your own Apps Script URL.
  • If you ever redeploy and get a new URL, update it here again.
